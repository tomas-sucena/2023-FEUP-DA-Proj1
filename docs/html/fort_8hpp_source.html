153 template <typename T>
154 constexpr bool is_stream_manipulator_impl() noexcept
155 {
156 using Tdec = typename std::decay<T>::type;
157 // Manipulators for integral types
158 return std::is_same<Tdec, typename std::decay<decltype(std::dec)>::type>::value
159 || std::is_same<Tdec, typename std::decay<decltype(std::hex)>::type>::value
160 || std::is_same<Tdec, typename std::decay<decltype(std::oct)>::type>::value
161 // Floating-point manipulators
162 || std::is_same<Tdec, typename std::decay<decltype(std::fixed)>::type>::value
163 || std::is_same<Tdec, typename std::decay<decltype(std::scientific)>::type>::value
164 // Misc
165 || std::is_same<Tdec, typename std::decay<decltype(std::setbase(0))>::type>::value
166 || std::is_same<Tdec, typename std::decay<decltype(std::setfill('\0'))>::type>::value
167 || std::is_same<Tdec, typename std::decay<decltype(std::setprecision(0))>::type>::value
168 || std::is_same<Tdec, typename std::decay<decltype(std::setw(0))>::type>::value;
169 }
170 }
171
177 template <typename T>
178 constexpr bool is_stream_manipulator() noexcept
179 {
180 return detail::is_stream_manipulator_impl<T>();
181 }

420 table()

423 {
424 if (table_ == NULL)
425 throw std::bad_alloc();

435
439 table(const table &tbl)
440 : property_owner_t(FT_ANY_ROW, FT_ANY_COLUMN, this), table_(NULL)
441 {
442 if (tbl.table_) {
443 ft_table_t *table_copy = ft_copy_table(tbl.table_);
444 if (table_copy == NULL)
445 throw std::runtime_error("Error during table copy");
446
447 stream_.str(std::string());
448 if (tbl.stream_.tellp() >= 0) {
449 stream_ << tbl.stream_.str();
450 }
451 table_ = table_copy;
452 }
453 }
454
458 table(table &&tbl)
459 : property_owner_t(FT_ANY_ROW, FT_ANY_COLUMN, this), table_(tbl.table_)
460 {
461 if (tbl.stream_.tellp() >= 0) {
462 stream_ << tbl.stream_.str();
463 tbl.stream_.str(std::string());
464 }
465 tbl.table_ = 0;
466 }
467
471 table &operator=(const table &tbl)
472 {
473 if (&tbl == this)
474 return *this;
475
476 if (tbl.table_) {
477 ft_table_t *table_copy = ft_copy_table(tbl.table_);
478 if (table_copy == NULL)
479 throw std::runtime_error("Error during table copy");
480
481 stream_.str(std::string());
482 if (tbl.stream_.tellp() >= 0) {
483 stream_ << tbl.stream_.str();
484 }
485 ft_destroy_table(table_);
486 table_ = table_copy;
487 }
488 return *this;
489 }
490
494 table &operator=(table &&tbl)
495 {
496 if (&tbl == this)
497 return *this;
498
499 if (tbl.table_) {
500 stream_.str(std::string());
501 if (tbl.stream_.tellp() >= 0) {
502 stream_ << tbl.stream_.str();
503 tbl.stream_.str(std::string());
504 }
505 ft_destroy_table(table_);
506 table_ = tbl.table_;
507 tbl.table_ = NULL;
508 }
509 return *this;
510 }

564 template <typename T>
565 table &operator<<(const T &arg)
566 {
567 constexpr bool is_manip = fort::is_stream_manipulator<typename std::decay<T>::type>();
568 stream_ << arg;
569 if (stream_.tellp() >= 0 && !is_manip) {
570#ifdef FT_HAVE_UTF8
571 if (TT == table_type::character) {
572 ft_nwrite(table_, 1, stream_.str().c_str());
573 } else {
574 ft_u8nwrite(table_, 1, (const void *)stream_.str().c_str());
575 }
576#else
577 ft_nwrite(table_, 1, stream_.str().c_str());
578#endif
579
580 stream_.str(std::string());
581 }
582 return *this;
583 }
584
585 table &operator<<(const table_manipulator &arg)
586 {
587 if (arg.value == header.value)
588 ft_set_cell_prop(table_, FT_CUR_ROW, FT_ANY_ROW, FT_CPROP_ROW_TYPE, FT_ROW_HEADER);
589 else if (arg.value == endr.value)
590 ft_ln(table_);
591 else if (arg.value == separator.value)
592 ft_add_separator(table_);
593 return *this;
594 }
